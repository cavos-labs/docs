## Setup

### 1. Configure the Provider

Wrap your application with `CavosProvider`:

```tsx
// app/layout.tsx or _app.tsx
import { CavosProvider } from '@cavos/react';

export default function RootLayout({ children }) {
  return (
    <CavosProvider
      config={{
        appId: 'your-app-id',
        network: 'sepolia', // or 'mainnet'
      }}
    >
      {children}
    </CavosProvider>
  );
}
```

### 2. Add Login Button

Use the `useCavos` hook to access authentication and wallet functions:

```tsx
'use client';
import { useCavos } from '@cavos/react';

export function LoginButton() {
  const { login, isAuthenticated, address, isLoading } = useCavos();

  if (isLoading) {
    return <button disabled>Loading...</button>;
  }

  if (isAuthenticated && address) {
    return (
      <div>
        Connected: {address.slice(0, 6)}...{address.slice(-4)}
      </div>
    );
  }

  return (
    <button onClick={() => login('google')}>
      Login with Google
    </button>
  );
}
```

### 3. Execute Transactions

Send gasless transactions with a single function call:

```tsx
import { useCavos } from '@cavos/react';

function TransferButton() {
  const { execute, address } = useCavos();

  const handleTransfer = async () => {
    const txHash = await execute(
      {
        contractAddress: '0x049d36570d4e46f48e99674bd3fcc84644ddd6b96f7c741b1562b82f9e004dc7',
        entrypoint: 'transfer',
        calldata: [recipientAddress, amount, '0'],
      },
      { gasless: true }
    );
    console.log('Transaction:', txHash);
  };

  return <button onClick={handleTransfer}>Send</button>;
}
```
